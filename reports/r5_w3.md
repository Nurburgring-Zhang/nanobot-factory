**修复后 (R5-W3)**:
```js
async function REVIEW_approveItem(id) {
  const review = REVIEW_STATE.reviews.find(r => r.id === id);
  if (!review) return;
  // 保存旧状态以便失败回滚
  const prevStatus = review.status;
  const prevReviewer = review.reviewer;
  const prevReviewedAt = review.reviewed_at;
  // 乐观更新 UI
  review.status = 'approved';
  review.reviewer = '当前用户';
  review.reviewed_at = new Date().toISOString().slice(0, 16).replace('T', ' ');
  REVIEW_STATE.selectedIds.delete(id);
  REVIEW_refreshUI();
  // 真实 API 调用 — 失败必须让用户知道
  try {
    const resp = await apiPost('/api/review/approve', { id });
    if (resp && resp.success === false) {
      throw new Error(resp.error || resp.message || '审核通过失败');
    }
    showToast('已通过审核 #' + id, 'success');
  } catch (e) {
    // 回滚 UI
    review.status = prevStatus;
    review.reviewer = prevReviewer;
    review.reviewed_at = prevReviewedAt;
    REVIEW_STATE.selectedIds.add(id);
    REVIEW_refreshUI();
    showToast('通过失败: ' + (e?.message || e), 'error');
  }
}
```

**改进** (4 项):
1. **错误不再静默**: 显式检查 `resp.success === false` 抛错 + catch 块
2. **失败时 UI 回滚**: 状态 / reviewer / reviewed_at 三项用 snapshot 恢复
3. **失败时重新加入 selectedIds**: 用户可再次操作同一项
4. **双向 toast 反馈**: 成功 → `'已通过审核 #id' success`; 失败 → `'通过失败: <msg>' error`

### 3.3 `review.js` — REVIEW_executeBatch (改: +60 / -10 行, 行 357-461)

**修复前 (R3 残留, 完全无 API 调用)**:
```js
function REVIEW_executeBatch(action, ids) {
  ids.forEach(id => {
    const review = REVIEW_STATE.reviews.find(r => r.id === id);
    if (!review) return;
    review.status = action === 'approve' ? 'approved' : action === 'reject' ? 'rejected' : 'skipped';
    review.reviewer = '当前用户(批量)';
    review.reviewed_at = new Date().toISOString().slice(0, 16).replace('T', ' ');
  });
  REVIEW_STATE.selectedIds = new Set();
  REVIEW_STATE.selectedId = null;

  REVIEW_refreshUI();

}
```

**修复后 (R5-W3, 真批量 + 失败回滚 + 3 种 toast)**:
```js
async function REVIEW_executeBatch(action, ids) {
  // 1. 防御: 空数组
  if (!Array.isArray(ids) || ids.length === 0) {
    showToast('未选择审核项', 'warning');
    return;
  }
  // 2. 快照旧状态 (失败回滚用)
  const snapshots = ids.map(id => { ... }).filter(Boolean);
  // 3. 乐观更新 UI (保留原行为)
  ids.forEach(id => { review.status = ...; review.reviewer = '当前用户(批量)'; ... });
  REVIEW_STATE.selectedIds = new Set();
  REVIEW_STATE.selectedId = null;
  closeModal();
  REVIEW_refreshUI();

  // 4. 真实批量 API — 双方案
  showGlobalLoading('批量' + actionLabel[action] + '中... ' + ids.length + ' 条');
  let results = null;
  // 4a. 方案 1: 尝试 batch 端点 (单次请求带 ids 数组)
  try {
    const batchResp = await apiPost('/api/review/' + actionPath + '-batch', { ids });
    if (batchResp && batchResp.success !== false) {
      results = ids.map(id => ({ id, ok: true }));
    } else if (batchResp && (batchResp.code === 404 || /not\s*found|404/i.test(...))) {
      results = null;  // 端点不存在, 走回退
    } else {
      results = null;  // 真实错误, 走回退
    }
  } catch (e) { results = null; }
  // 4b. 方案 2: 回退到并发单条 (类似 sc_triggerAll 模式)
  if (!results) {
    const settled = await Promise.allSettled(
      ids.map(id => apiPost('/api/review/' + actionPath, { id }))
    );
    results = settled.map((s, i) => ({
      id: ids[i],
      ok: s.status === 'fulfilled' && s.value && s.value.success !== false,
      err: s.status === 'fulfilled'
        ? (s.value?.error || s.value?.message || '失败')
        : (s.reason?.message || String(s.reason || '失败')),
    }));
  }
  hideGlobalLoading();

  // 5. 失败项回滚 UI
  if (failList.length > 0) {
    failList.forEach(fail => { /* 用 snapshots 恢复 + 加回 selectedIds */ });
    REVIEW_refreshUI();
  }

  // 6. 三种 toast: 全成功 / 全失败 / 部分失败
  if (failList.length === 0) {
    showToast('已批量' + actionLabel[action] + ' ' + okList.length + ' 条', 'success');
  } else if (okList.length === 0) {
    showToast('批量' + actionLabel[action] + '失败: ' + failList[0].err, 'error');
  } else {
    showToast('部分失败: 成功 ' + okList.length + ', 失败 ' + failList.length +
              ' (首个: ' + (failList[0].err || '未知') + ')', 'warning');
  }
}
```

**设计要点** (5 项):
1. **乐观 UI + 真实 API + 失败回滚** 三段式, 与 approve/reject 单条保持一致
2. **优先 batch 端点**: 未来后端加 `/api/review/{action}-batch` 时自动用上 (减少 N 次 HTTP → 1 次)
3. **回退到并发单条**: 即使 batch 端点不存在也不静默失败, 像 `sc_triggerAll` 那样并发调 N 次单条
4. **全成功/全失败/部分失败** 三种 toast, 部分失败时显示首个错误原因
5. **失败项 UI 回滚 + 重新加 selectedIds**: 用户可以再操作失败的项

## 7. 已知瑕疵 (留 R6 / 后续)

1. **review.js 端点语义不匹配**: 当前 `/api/review/approve` 是算法审核端点 (algo_id), 不是标注审核。标注审核应在 `/api/quality/v2/review/process` (quality_v2_routes.py:565, payload `{item_id, reviewer_id, decision, comments}`)。
   - **影响**: 标注审核调用会 4xx, 但 R5-W3 修复后用户至少能看到错误提示
   - **修复建议**: R6 UX 阶段把 review.js 的端点改到 `/api/quality/v2/review/process`
   - **本轮范围**: 限定 "错误不吞掉 + 真批量 API", 不改端点路径

2. **sc_triggerAll 端点选择**: 任务描述 `/api/scheduler/trigger-all` 但后端无此端点, 当前并发循环是正确替代
   - **影响**: 无 (功能等价)
   - **修复建议**: 未来加 `/api/scheduler/trigger-all` batch 端点时, 改用类似 REVIEW_executeBatch 的双方案模式

3. **renderReview 加载端点错配**: `apiGet('/api/review/queue')` 不存在 (应是 `/api/review/` 或 `/api/quality/v2/review/queue-stats`)
   - **影响**: 审核列表加载为空 (但 fallback 到空数组, UI 正常显示 "审核队列已清空")
   - **修复建议**: R6 修

4. **REVIEW_applyFilters 重渲染丢 checkbox 选中状态**: 重新 `innerHTML = REVIEW_renderQueue(filtered)` 会丢失所有 checkbox 选中
   - **影响**: 选中 → 搜索 → 选中项丢失
   - **修复建议**: R6 改用 diff patching 或保留 selectedIds + 重新绑定 event

5. **REVIEW_refreshUI 闪烁**: 选中详情后, `REVIEW_refreshUI` 内部 `REVIEW_applyFilters` 重渲染整个列表, 然后 `REVIEW_selectItem` 恢复 detail
   - **影响**: 视觉闪烁
   - **修复建议**: R6 改用局部更新

6. **sc_filter / sc_filterByStatus 输入抖动未做 debounce**: 用户每键入一次就过滤一次
   - **影响**: 大列表 (100+ 任务) 时性能略差
   - **修复建议**: R6 加 200ms debounce
